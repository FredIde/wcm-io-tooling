53    @Test
54    public void testServiceRegistration() throws InvalidSyntaxException {
55      // prepare test services
56      String clazz1 = String.class.getName();
57      Object service1 = new Object();
58      Dictionary properties1 = getServiceProperties(null);
59      ServiceRegistration reg1 = this.bundleContext.registerService(clazz1, service1, properties1);
60  
61      String[] clazzes2 = new String[] {
62          String.class.getName(), Integer.class.getName()
63      };
64      Object service2 = new Object();
65      Dictionary properties2 = getServiceProperties(null);
66      ServiceRegistration reg2 = this.bundleContext.registerService(clazzes2, service2, properties2);
67  
68      String clazz3 = Integer.class.getName();
69      Object service3 = new Object();
70      Dictionary properties3 = getServiceProperties(100L);
71      ServiceRegistration reg3 = this.bundleContext.registerService(clazz3, service3, properties3);
72  
73      // test get service references
74      ServiceReference refString = this.bundleContext.getServiceReference(String.class.getName());
75      assertSame(reg1.getReference(), refString);
76  
77      ServiceReference refInteger = this.bundleContext.getServiceReference(Integer.class.getName());
78      assertSame(reg3.getReference(), refInteger);
79  
80      ServiceReference[] refsString = this.bundleContext.getServiceReferences(String.class.getName(), null);
81      assertEquals(2, refsString.length);
82      assertSame(reg1.getReference(), refsString[0]);
83      assertSame(reg2.getReference(), refsString[1]);
84  
85      ServiceReference[] refsInteger = this.bundleContext.getServiceReferences(Integer.class.getName(), null);
86      assertEquals(2, refsInteger.length);
87      assertSame(reg3.getReference(), refsInteger[0]);
88      assertSame(reg2.getReference(), refsInteger[1]);
89  
90      ServiceReference[] allRefsString = this.bundleContext.getAllServiceReferences(String.class.getName(), null);
91      assertArrayEquals(refsString, allRefsString);
92  
93      // test get services
94      assertSame(service1, this.bundleContext.getService(refsString[0]));
95      assertSame(service2, this.bundleContext.getService(refsString[1]));
96      assertSame(service3, this.bundleContext.getService(refInteger));
97  
98      // unget does nothing
99      this.bundleContext.ungetService(refsString[0]);
100     this.bundleContext.ungetService(refsString[1]);
101     this.bundleContext.ungetService(refInteger);
102   }
103 
104   private Dictionary getServiceProperties(final Long serviceRanking) {
105     Dictionary<String, Object> props = new Hashtable<>();
106     if (serviceRanking != null) {
107       props.put(Constants.SERVICE_RANKING, serviceRanking);
108     }
109     return props;
110   }
